class EgybestTest extends StatefulWidget {
  const EgybestTest({super.key});
  @override
  State<EgybestTest> createState() => _EgybestTestState();
}

class _EgybestTestState extends State<EgybestTest> {
  final TextEditingController _urlController = TextEditingController(text: "kbkp2ostjlha");
  BetterPlayerController? _betterPlayerController;
  HlsProxyServer? _proxy;
  List<DlnaDevice> devices = [];
  DlnaDevice? selectedDevice;
  String? proxyBaseUrl;
  late DlnaService dlnaService;

@override
void initState() {
  super.initState();
   dlnaService =  Provider.of<DlnaProvider>(context, listen: false).dlnaService;
}

  Future<void> _initProxy() async {
    if (_proxy != null) return;
    final ip = await getLocalIp();
    debugPrint(ip);
    proxyBaseUrl = "http://$ip:8080/master.m3u8";
    _proxy = HlsProxyServer(headers: egybestHeaders, localIp: ip);
    await _proxy!.start(port: 8080);
  }


  void playOnBetterPlayer() async {
    await _initProxy();
    final id = _urlController.text;
   await dlnaService.playWithTitle("$proxyBaseUrl?id=$id", "movie1 1");
    return;
    final dataSource = BetterPlayerDataSource(
      BetterPlayerDataSourceType.network,title: "test egy best",
      "$proxyBaseUrl?id=$id",
    );

    _betterPlayerController = BetterPlayerController(
      const BetterPlayerConfiguration(autoPlay: true),
      betterPlayerDataSource: dataSource,
    );

    Navigator.push(context, MaterialPageRoute(builder: (_) => PlayerPage(controller: _betterPlayerController!)));
  }

  @override
  Widget build(BuildContext context) {
    return Scaffold(
      appBar: AppBar(title: const Text("HLS Proxy Player")),
      body: Padding(
        padding: const EdgeInsets.all(12),
        child: Column(
          children: [
            TextField(controller: _urlController, decoration: const InputDecoration(labelText: "Movie ID", border: OutlineInputBorder())),
            const SizedBox(height: 12),
            ElevatedButton(onPressed: playOnBetterPlayer, child: const Text("Play on BetterPlayer")),
          ],
        ),
      ),
    );
  }
}

class PlayerPage extends StatelessWidget {
  final BetterPlayerController controller;
  const PlayerPage({super.key, required this.controller});

  @override
  Widget build(BuildContext context) {
    return Scaffold(
      appBar: AppBar(title: const Text("BetterPlayer")),
      body: BetterPlayer(controller: controller),
    );
  }
}
